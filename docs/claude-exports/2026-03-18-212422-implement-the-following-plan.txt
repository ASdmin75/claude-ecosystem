│ Контекст                                                                     │
│                                                                              │
│ Проект содержит 13 MCP-серверов (cmd/mcp/mcp-*), каждый из которых вручную   │
│ реализует JSON-RPC 2.0 stdio протокол. Идентичный boilerplate (~60-80 строк) │
│  дублируется в каждом сервере: типы jsonRPCRequest/jsonRPCResponse,          │
│ структура tool, main-loop с bufio.Scanner, обработка                         │
│ initialize/tools/list/tools/call. Вопрос — стоит ли заменить этот            │
│ boilerplate на библиотеку mcp-go.                                            │
│                                                                              │
│ Вердикт: Да, имеет смысл                                                     │
│                                                                              │
│ Аргументы "за"                                                               │
│                                                                              │
│ 1. Устранение дублирования — 13 серверов × ~70 строк boilerplate = ~900      │
│ строк копипасты. mcp-go сводит инициализацию к 3-5 строкам (NewMCPServer +   │
│ ServeStdio).                                                                 │
│ 2. Корректность протокола — Ручная реализация не обрабатывает edge cases     │
│ (notifications без id, notifications/initialized, ping). mcp-go гарантирует  │
│ полное соответствие спецификации.                                            │
│ 3. Совместимость с будущими версиями MCP — Текущий код жёстко зашит под      │
│ 2024-11-05. mcp-go уже поддерживает 2025-11-25 с обратной совместимостью.    │
│ Обновление протокола — просто go get -u.                                     │
│ 4. Типобезопасное извлечение аргументов — req.RequireString("name") вместо   │
│ ручного json.Unmarshal + type assertions через map[string]any.               │
│ 5. Разблокирует новые возможности — Resources, Prompts, Middleware, async    │
│ Tasks — всё доступно без написания протокольного кода. Например, middleware  │
│ для логирования/метрик можно добавить один раз.                              │
│ 6. Минимальные зависимости — Библиотека тянет всего 5 прямых зависимостей    │
│ (uuid, jsonschema, cast, testify, uritemplate).                              │
│                                                                              │
│ Аргументы "против" (и почему они не критичны)                                │
│                                                                              │
│ ┌───────────────────────────┬─────────────────────────────────────────────┐  │

│ // 25 строк типов jsonRPC                                                    │

│ // handler функции                                                           │

│ После (~200 строк):                                                          │
│ func main() {                                                                │
│     s := server.NewMCPServer("filesystem", "0.1.0")                          │
│     s.AddTool(mcp.NewTool("read_file",                                       │
│         mcp.WithDescription("Read file contents"),                           │
│         mcp.WithString("path", mcp.Required(), mcp.Description("File         │
│ path")),                                                                     │
│     ), handleReadFile)                                                       │
│     // ... остальные tools                                                   │
│     server.ServeStdio(s)                                                     │
│ }                                                                            │
│                                                                              │
│ func handleReadFile(ctx context.Context, req mcp.CallToolRequest)            │
│ (*mcp.CallToolResult, error) {                                               │
│     path, _ := req.RequireString("path")                                     │
│     data, err := os.ReadFile(path)                                           │
│     if err != nil {                                                          │
│         return mcp.NewToolResultError(err.Error()), nil                      │
│     }                                                                        │
│     return mcp.NewToolResultText(string(data)), nil                          │
│ }                                                                            │
